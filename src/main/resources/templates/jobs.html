async function applyForJob(jobId) { try { const response = await fetch('/api/applications', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ jobId: jobId }) }); if (!response.ok) { const error = await response.json(); throw new Error(error.error || 'Failed to apply for job'); } alert('Application submitted successfully!'); window.location.reload(); } catch (error) { console.error('Error applying for job:', error); alert(error.message); } }
